    <script>

        let cart = [];

        const buttons = document.querySelectorAll('.btn');

        const cartLink = document.getElementById('cartLink');

        const cartContent = document.getElementById('cartContent');

        // Add to cart functionality

        buttons.forEach((button, index) => {

            button.addEventListener('click', () => {

                let productName = button.parentElement.querySelector('h2').textContent;

                let productPrice = button.parentElement.querySelector('.price').textContent;

                cart.push({ name: productName, price: productPrice });

                updateCart();

                alert(productName + ' added to cart!');

            });

        });

        function updateCart() {

            cartLink.textContent = 'Cart (' + cart.length + ')';

            cartContent.innerHTML = cart.length ? cart.map(item => `<p>${item.name} - ${item.price}</p>`).join('') : 'No items in cart.';

        }

        // Checkout Simulation

        function checkout() {

            if(cart.length) {

                alert('This is a demo checkout process. No real transaction will occur.');

                cart = [];

                updateCart();

            } else {

                alert('Your cart is empty.');

            }

        }
